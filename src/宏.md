# 不定参数宏示例
```c
typedef struct tmp
{
  int a;
  int b;
} tmp_t;
tmp_t data;
char *str = "string."

#define DATA_DEBUG(data, fmt, ...) \
  DATA_PRINT(DEBUG, data, fmt, ##__VA_ARGS__)

#define DATA_PRINT(level, d, fmt, ...) \
do {
  __DATA_PRINT("%d, %d"fmt, \
  d.a, d.b, \
  ##__VA_ARGS__);
} while(0)


// 函数中使用
DATA_DEBUG(data, "something %s\n", str);
```
